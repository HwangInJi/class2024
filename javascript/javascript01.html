01. 변수 : 데이터 저장

let을 사용하여 x, y, z에 변수값을 입력합니다.
변수는 데이터를 저장하는 기능을 하며, 이 때 저장된 데이터는 재사용이 가능합니다. 

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    console.log(x);
    console.log(y);
    console.log(z);
}
결과 확인하기
100
200
javascript

02. 변수 : 데이터 저장 + 데이터 변경

변수값 x는 초기값인 100에서 300으로, y는 200에서 400으로, z는 "javascript"에서 "jquery"로 변경합니다. 

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    x = 300;
    y = 400;
    z = "jquery";

    console.log(x);
    console.log(y);
    console.log(z);
}
결과 확인하기
300
400
jquery

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

+를 사용하여 변수값 x에는 300, y에는 400, z에는 "jquery"를 추가합니다.
이 때, 변수값은 변경과 추가가 가능합니다. 

{
    let x = 100;
    let y = 200;
    let z = "javascript";

    x += 300;
    y += 400;
    z += "jquery";

    console.log("03_________");
    console.log(x);
    console.log(y);
    console.log(z);
}
결과 확인하기
400
600

12. 배열 : 데이터 저장(여러개) : 표현방법 8 : 배열펼침연산자

배열 선언을 통해 각각의 arr1, arr2, arr3값을 지정합니다.
console.log()를 이용해 결과를 얻을 때 arr앞에 ...이 붙은 경우엔 결과값은 ,가 붙지 않은 상태로 도출됩니다. 

{
    const arr1 = [100, 200];
    const arr2 = [300, 400];
    const merge1 = [...arr1, ...arr2];
    const merge2 = [...arr1, "javascript", ...arr2, "jquery"];

    console.log(arr1); //[100, 200]
    console.log(...arr1); // 100, 200
    console.log(...arr1, ...arr2);      // ...이 붙으면 배열과 , 생략
    console.log(arr1.concat(arr2));     // concat을 사용하여 배열로 합침
    console.log(merge1);                // arr을 merge1로 지정하면 배열로 합쳐짐
    console.log(merge2);                // arr을 merge2로 지정하면 배열로 합쳐지고 문자는 ''로 도출
}
결과 확인하기
[100, 200]
100, 200
100, 200, 300, 400
[100, 200, 300, 400]
[100, 200, 300, 400]
[100, 200, 'javascript', 300, 400, 'jquery']

17. 객체 : 데이터 저장(키와값) : 표현방법 5 : 배열 안에 객체가 있는 방식

1. 각각의 []와 {}는 하나의 덩어리로 묶어 큰 덩어리인 []값 안에 두 개의 작은 덩어리인 {}값이 있음을 먼저 생각해야 합니다.
2. 큰 덩어리 []값은 두 개이므로 배열 요소는 [0], [1]이 됩니다.
3. 첫 번째 작은 덩어리 {}값에서 객체 속성은 총 3개이며, 첫번째 객체 속성.a는 100, 두번째 객체 속성.b는 200, 세번째 객체 속성.c는 "javascript"가 됩니다.
4. 두 번째 작은 덩어리 {}값에서 객체 속성은 총 3개이며, 첫번째 객체 속성.a는 300, 두번째 객체 속성.b는 400, 세번째 객체 속성.c는 "jquery"가 됩니다.
5. console.log()를 사용하여 객체값에 배열 요소와 객체 속성을 순서대로 입력하면 결과에 해당되는 초기값이 출력됩니다.

{

            b: 200,

        }, {

            b: 400,
            c: "jquery",
        }

    console.log(obj[0].a);  
    console.log(obj[0].b);  
    console.log(obj[0].c);  
    console.log(obj[1].a);

++ 연산자(논리, 산술, 비교, 조건...)

1. 산술 연산자


① 덧셈(+) - sum : 두 개 이상의 숫자나 문자열을 더할 때 사용

{
    let sum = 5 + 3;
    console.log(sum); 

    let concatenation = "Hello " + "world"; 
    console.log(concatenation); 
}

② 뺄셈(-) - difference : 두 숫자간의 차이를 계산할 때 사용

{
    let difference = 10 - 4;
    console.log(difference);
}

③ 곱셈(*) - product : 두 숫자의 곱을 계산할 때 사용

{
    let product = 6 * 4;
    console.log(product);
}

④ 나눗셈(/) - quotient : 두 숫자를 나눌 때 사용되며, 나눗셈의 결과는 부동 소수점 형태로 반환될 수 있음

{
    let quotient = 10 / 2;
    console.log(quotient);
}

⑤ 나머지(%) - remainder : 두 숫자를 나눈 후 남는 나머지를 계산할 때 사용

{
    let remainder = 10 % 3;
    console.log(remainder);
}

2. 논리 연산자


① AND(&&) : 두 조건이 모두 참일 때만 전체 표현식이 참이 되는 경우

② OR(||) : 두 조건 중 하나 이상이 참이 되는 경우

③ NOT(!) : 주어진 조건의 결과를 반대를 반환

3. 비교 연산자


① 같음(==, ===) : 두 조건이 모두 참일 때만 전체 표현식이 참이 되는 경우

- == : 연산자의 두 값이 동등한지 여부를 확인하며, 자료형이 다르더라도 일치하면 true를 반환

{
    let num1 = 5;
    let num2 = "5";

    console.log(num1 == num2);  // true (값만 비교하므로 true)
    console.log(num1 === num2); // false (값과 자료형이 다르므로 false)
}

- === : 연산자의 두 값과 자료형이 모두 동일한지 확인하며, 모두 일치해야만 true를 반환

{
    let num1 = 5;
    let num2 = "5";

    console.log(num1 != num2);  // false (값은 같으므로 false)
    console.log(num1 !== num2); // true (값은 같지만 자료형이 다르므로 true)
}

② 다름(!=, !==) : 두 조건 중 하나 이상이 참이 되는 경우

- != : 연산자의 두 값이 서로 다른지 여부를 확인하며, 자료형이 다르더라도 값이 다르면 true를 반환

{
let num1 = 5;
let num2 = "5";

console.log(num1 == num2);  // true (값만 비교하므로 true)
console.log(num1 === num2); // false (값과 자료형이 다르므로 false)
}

- !== : 연산자의 두 값과 자료형이 모두 다른지 확인하며, 값과 자료형이 하나라도 다르면 true를 반환

{
let num1 = 5;
let num2 = "5";

console.log(num1 != num2);  // false (값은 같으므로 false)
console.log(num1 !== num2); // true (값은 같지만 자료형이 다르므로 true)
}

③ 크다(>) : 숫자 비교에 많이 사용되며, 왼쪽 피연산자가 오른쪽 피연산자보다 큰지 여부를 확인

{
    let num1 = 5;
    let num2 = 10;

    console.log(num2 > num1); // true (10은 5보다 큼)
}

④ 작다(<) : 숫자 비교에 많이 사용되며, 왼쪽 피연산자가 오른쪽 피연산자보다 작은지 여부를 확인

{
    let num1 = 5;
    let num2 = 10;

    console.log(num1 < num2); // true (5는 10보다 작음)
}

⑤ 크거나 같다(>=) : 왼쪽 피연산자가 오른쪽 피연산자보다 크거나 같은지 여부를 확인

{
    let num1 = 5;
    let num2 = 10;

    console.log(num2 >= num1); // true (10은 5보다 크거나 같음)
}

⑥ 작거나 같다(<=) : 왼쪽 피연산자가 오른쪽 피연산자보다 작거나 같은지 여부를 확인

{
    let num1 = 5;
    let num2 = 10;

    console.log(num1 <= num2); // true (5는 10보다 작거나 같음)
}
